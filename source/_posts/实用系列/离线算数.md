### 草稿

这应该是一种类似于 HTML 的“标记型语言”：

| 符号        | 意思                         | C                              | Python                       |
| ----------- | ---------------------------- | ------------------------------ | ---------------------------- |
| $k$         | 关键点的位置                 | 传入函数的参数                 | 同                           |
| $.$         | 当前指针的位置               | i                              | 同                           |
| $-$         | 当前指针的左边               | i - 1                          | 同                           |
| $+$         | 当前指针的右边               | i + 1                          | 同                           |
| $0$         | 第一个元素的位置             | 0                              | 同                           |
| $1$         | 第一个元素右边               | 1                              | 同                           |
| $p$         | 最后一个元素左边             | len - 2                        | len - 2 或 -2                |
| $e$         | 最后一个元素的位置           | len - 1                        | len - 1 或 -1                |
| $q$         | 最后一个元素右边             | len                            | 同                           |
| $7_{a}^{b}$ | 从下标正序遍历到上标         | for(int i = a; i < b + 1; i++) | for i in range(a, b + 1)     |
| $L_{a}^{b}$ | 从下标倒序遍历到上标         | for(int i = a; i >= b; i--)    | for i in range(a, b - 1, -1) |
| $<$         | 赋值                         | =                              | 同                           |
| $+<.$       | 把当前指针指向的赋值给右边的 | arr[i+1] = arr[i]              | 同                           |
| $B$         | 判断索引是否越界             |                                |                              |
| $F$         | 判断空间是否已满             |                                |                              |
| $?$         | 判断值是否相等               |                                |                              |
| $N$         | 改变数据结构的长度           |                                |                              |

例如，$L_{e}^{k}(+<.)$表示：

指针从最后一个元素倒序遍历到关键点，在每次循环中把指针指向的元素赋值给指针右边的元素。

```c
for(int i = length - 1; i >= k; i--)
{
    arr[i+1] = arr[i];
}
```

### 实例

遍历二维数组：$7_0^e 7_0^e$

遍历矩阵主对角线及其以上的部分：$7_{i=0}^e 7_{j=i}^e$

顺序表里根据索引取值：$B$

顺序表里根据值查找索引：$7_{0}^{e}(?)$

插入元素到顺序表：$BFL_{e}^{k}(+<.)N$

删除顺序表里的某个元素：$B7_{k}^{e}(.<+)N$

## 图的邻接方阵

行索引 i 为所有节点的排列。列索引 j 与其相同。

矩阵的主对角线上全为 0，以主对角线轴对称。

(i,j) 或 (j,i) 为 1 时，代表两个节点连上线了。

同构的两个图，它们的邻接方阵是等价的。反之不一定。因为：

1. 一个图的邻接方阵，只能做行/列对换操作。且在对换两行之后，必须要对换相同编号的两列，才能保证同构。
2. 等价矩阵的定义是：经过有限次初等变换后的矩阵。

考虑这样的两个图：

$$
\begin{pmatrix}
0 & 1 & 0 \\
1 & 0 & 0 \\
0 & 0 & 0 \\
\end{pmatrix}
$$

与

$$
\begin{pmatrix}
0 & 1 & 1 \\
1 & 0 & 0 \\
1 & 0 & 0 \\
\end{pmatrix}
$$

它们的邻接方阵是等价的，但是前者是两个点连线加上一个孤立点，后者是三个点连在一条线上（无环），显然不同构。

void printMatrix(int mat[R][C])
{
    for (int i = 0; i < R; i++)
    {
        for (int j = 0; j < C; j++)
        {
            printf("%d\t", mat[i][j]);
        }
        printf("\n");

void swapRows(int mat[R][C], int r1, int r2)
{
    // 传的是行索引，变的是列索引
    for (int j = 0; j < C; j++)
    {
        int temp = mat[r1][j];
        mat[r1][j] = mat[r2][j];
        mat[r2][j] = temp;
    }
}

void swapCols(int mat[R][C], int c1, int c2)
{
    // 传的是列索引，变的是行索引
    for (int i = 0; i < R; i++)
    {
        int temp = mat[i][c1];
        mat[i][c1] = mat[i][c2];
        mat[i][c2] = temp;
    }
}

void rMultiPlyK(int mat[R][C], int r, int k)
{
    for (int j = 0; j < C; j++)
    {
        mat[r][j] *= k;
    }
}

void cMultiPlyK(int mat[R][C], int c, int k)
{
    for (int i = 0; i < R; i++)
    {
        mat[i][c] *= k;
    }
}

void r2AddAssignR1MultiPlyK(int mat[R][C], int r2, int r1, int k)
{
    for (int j = 0; j < C; j++)
    {
        mat[r2][j] += mat[r1][j] * k;
    }
}

void c2AddAssignC1MultiPlyK(int mat[R][C], int c2, int c1, int k)
{
    for (int i = 0; i < R; i++)
    {
        mat[i][c2] += mat[i][c1] * k;
    }
}

## 幂集

集合的所有子集构成的集合。

$A$ 的幂集记作 $2^A$。幂集里元素的个数等于 $2^{原集合里元素的个数}$

```py
def powerset(s: set) -> list[set]:
    """
    幂集，一个集合的所有子集。这里用 list，因为 set 里的东西是不可变的，但是 set 本身是可变的。所以不能 set 套 set
    """
    return [
        set(combination)
        for r in range(len(s) + 1)
        for combination in itertools.combinations(s, r)
    ]
```

## 顺序表

```c
#include <stdio.h>
#include <stdlib.h>

typedef int ElemType;

#define MAXSIZE 3 // 分配给顺序表的空间
typedef struct    // 顺序表
{
    ElemType *elem; // 基地址
    int length;     // 当前长度
} SqList;

void initList(SqList *L);
ElemType getElem(SqList *L, int i);
int locateElem(SqList *L, ElemType e);
void listInsert(SqList *L, int i, ElemType e);
void listDelete(SqList *L, int i);
void printList(SqList *L);

int main()
{
    SqList L;
    initList(&L);
    listInsert(&L, 0, 1);
    listInsert(&L, 0, 2);
    listInsert(&L, 2, 3);
    listInsert(&L, 0, 4);
    printList(&L);
    listDelete(&L, 0);
    printList(&L);
}

void printList(SqList *L)
{
    for (int i = 0; i < L->length; i++)
    {
        printf("%d,", L->elem[i]);
    }
    printf("\n");
}

void initList(SqList *L)
{
    // 给顺序表分配空间
    L->elem = (ElemType *)malloc(MAXSIZE * sizeof(ElemType));
    if (!L->elem)
    {
        exit(-1);
    }
    L->length = 0;
    printf("init success\n");
}

int getElem(SqList *L, int i)
{
    // 根据索引取值
    // 只要输入索引，就要校验是否合法B
    // 时间复杂度O(1)
    if (i < 0 || i > L->length - 1)
    {
        return -1;
    }
    return L->elem[i];
}

int locateElem(SqList *L, ElemType e)
{
    // 根据值查找索引，返回第一个相等值的索引。
    // 70e(?)
    // 时间复杂度O(n)
    for (int i = 0; i < L->length; i++)
    {
        if (e == L->elem[i])
        {
            return i;
        }
    }
    return -1;
}

void listInsert(SqList *L, int i, ElemType e)
{
    // 按索引插入元素，索引之后的元素向后移动。
    // BFLek(+<.)N
    // 这里输入索引了，所以要校验是否合法。由于是插入，还要检查空间是否已满。
    // 时间复杂度O(n)
    if (i < 0 || i > L->length)
    {
        printf("index out of bounds\n");
        return;
    }
    if (MAXSIZE == L->length)
    {
        printf("list is full\n");
        return;
    }

    for (int j = L->length - 1; j >= i; j--)
    {
        L->elem[j + 1] = L->elem[j];
    }
    L->elem[i] = e;
    L->length++;
    printf("insert success\n");
}

void listDelete(SqList *L, int i)
{
    // 按索引删除元素。
    // B7ke(.<+)N
    if (i < 0 || i > L->length - 1)
    {
        printf("index out of bounds\n");
        return;
    }
    for (int j = i; j < L->length - 1; j++)
    {
        L->elem[j] = L->elem[j + 1];
    }
    L->length--;
    printf("delete success\n");
}
```
